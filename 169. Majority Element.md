class Solution {
    public int majorityElement(int[] nums) {
        int mjCounter = 1;
        int mjElement = nums[0];
        
        for (int i = 1; i < nums.length; i++) {
            if (mjCounter == 0) {
                mjCounter = 1;
                mjElement = nums[i];
            } else if (nums[i] == mjElement) {
                mjCounter += 1;
            } else {
                mjCounter -= 1;
            }
        }
        
        return mjElement;
    }
}
